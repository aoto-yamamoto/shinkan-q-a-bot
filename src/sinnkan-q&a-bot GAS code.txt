// Webアプリとして開かれたときにindex.htmlを返す
// メンテナンスモードの旗を画面に埋め込んで渡す（HA-1）
function doGet() {
  const template = HtmlService.createTemplateFromFile('index');
  template.maintenance = isMaintenance() ? 'true' : 'false';
  return template.evaluate()
    .setTitle('新歓Q&A')
    .addMetaTag('viewport', 'width=device-width, initial-scale=1');
}

// スクリプトプロパティ MAINTENANCE が 'true' のとき受付停止（HA-1）
// 停止方法: プロジェクトの設定 > スクリプトプロパティに MAINTENANCE = true を追加
// 再開方法: 値を false にするかプロパティを削除
function isMaintenance() {
  return PropertiesService.getScriptProperties().getProperty('MAINTENANCE') === 'true';
}

// フロントから呼ばれる入口
// sessionId: ブラウザ側で生成したセッション識別子（レート制限用）
function handleQuestion(question, sessionId) {
  // メンテナンス中は受け付けない（画面表示後に停止した場合の保険）
  if (isMaintenance()) {
    return { id: null, reply: '現在質問の受付を一時停止中です。質問がある方はお手数ですが、公式LINEからお願いいたします。', escalated: false, limited: true };
  }

  // 手順1: レート制限チェック（利用上限）
  if (!checkRateLimit(sessionId)) {
    return { id: null, reply: RATE_LIMIT_MSG, escalated: false, limited: true };
  }

  // 手順2: 入力検証（KA-4のうちサーバ側検証）
  if (!question || typeof question !== 'string') {
    throw new Error('質問が空です');
  }
  question = question.trim();
  if (question.length === 0 || question.length > 500) {
    throw new Error('質問は1〜500文字で入力してください');
  }

  // 手順3: 記録（KA-5）
  const id = recordInquiry(question);

  // 手順4〜5: 意図分類＋検証（KA-4, KA-6）
  const faqId = classifyQuestion(question);

  // 手順6: 回答決定と行の更新（KA-7）
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const row = findRowById(id);
  let reply, status;
  if (faqId >= 1 && faqId <= 6) {
    reply = FAQ[faqId];
    status = '自動回答済';
  } else {
    reply = ESCALATION_MSG;
    status = '未対応';
  }
  sheet.getRange(row, 6).setValue(faqId === 0 ? '非該当' : faqId); // F: 判定結果
  sheet.getRange(row, 7).setValue(reply);                          // G: 自動返信内容
  sheet.getRange(row, 8).setValue(status);                         // H: ステータス

  return { id: id, reply: reply, escalated: (status === '未対応'), limited: false };
}

// レート制限: CacheServiceでセッションごとの回数を数える（1時間で自動リセット）
// 戻り値: true = 送信を許可 / false = 上限超過
function checkRateLimit(sessionId) {
  if (!sessionId || typeof sessionId !== 'string') return false; // ID無しは拒否
  const cache = CacheService.getScriptCache();
  const key = 'rate_' + sessionId.slice(0, 60);
  const count = Number(cache.get(key) || 0);
  if (count >= RATE_LIMIT_PER_HOUR) return false;
  cache.put(key, String(count + 1), 3600); // 3600秒 = 1時間保持
  return true;
}

// KA-5: 問い合わせ内容をスプレッドシートに記録する
function recordInquiry(question) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const lastRow = sheet.getLastRow();
  const newId = (lastRow <= 1) ? 1 : sheet.getRange(lastRow, 1).getValue() + 1;
  sheet.appendRow([newId, new Date(), '', '', question, '', '', '処理中', '', '']);
  return newId;
}

// 受付IDから行番号を特定する
function findRowById(id) {
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const ids = sheet.getRange(2, 1, Math.max(sheet.getLastRow() - 1, 1), 1).getValues();
  for (let i = 0; i < ids.length; i++) {
    if (ids[i][0] === id) return i + 2;
  }
  throw new Error('受付ID ' + id + ' が見つかりません');
}

// KA-8: 受付IDに紐づく行に氏名・メールアドレスを追記する
function addContactInfo(id, email, name) {
  if (!email || typeof email !== 'string' || email.indexOf('@') === -1 || email.length > 100) {
    throw new Error('メールアドレスの形式が正しくありません');
  }
  name = (typeof name === 'string') ? name.trim().slice(0, 50) : '';
  const sheet = SpreadsheetApp.getActiveSpreadsheet().getSheetByName(SHEET_NAME);
  const row = findRowById(id);
  sheet.getRange(row, 3).setValue(name);
  sheet.getRange(row, 4).setValue(email.trim());
  return true;
}

// 意図分類まわり（KA-4, KA-6）
// ============================================================

function classifyQuestion(question) {
  if (!USE_REAL_API) {
    return classifyStub(question);
  }

  const apiKey = PropertiesService.getScriptProperties().getProperty('ANTHROPIC_API_KEY');
  const payload = {
    model: 'claude-haiku-4-5-20251001',
    max_tokens: 50,
    system: SYSTEM_PROMPT,
    messages: [{ role: 'user', content: question }]
  };

  try {
    const res = UrlFetchApp.fetch('https://api.anthropic.com/v1/messages', {
      method: 'post',
      contentType: 'application/json',
      headers: { 'x-api-key': apiKey, 'anthropic-version': '2023-06-01' },
      payload: JSON.stringify(payload),
      muteHttpExceptions: true
    });

    if (res.getResponseCode() !== 200) {
      Logger.log('APIエラー: ' + res.getContentText());
      return 0;
    }

    const text = JSON.parse(res.getContentText()).content
      .filter(function(b) { return b.type === 'text'; })
      .map(function(b) { return b.text; })
      .join('');

    return parseFaqId(text);
  } catch (e) {
    Logger.log('API呼び出し失敗: ' + e);
    return 0;
  }
}

// スタブ: API連携前のテスト用

  Utilities.sleep(2000); // 本物のAPIの応答時間を模擬
  if (question.indexOf('持ち物') !== -1) return 3;
  if (question.indexOf('服') !== -1) return 5;
  if (question.indexOf('開催日') !== -1 || question.indexOf('いつ') !== -1) return 1;
  if (question.indexOf('何時') !== -1) return 6;
  if (question.indexOf('懇親会') !== -1) return 4;
  if (question.indexOf('申し込み') !== -1) return 2;
  if (question.indexOf('エラー') !== -1) throw new Error('スタブ: API障害を模擬');
  return 0;
}

// KA-6の検証（範囲チェック）を含むJSON解析
function parseFaqId(rawText) {
  try {
    const id = Number(JSON.parse(rawText.trim()).faq_id);
    if (Number.isInteger(id) && id >= 0 && id <= 6) return id;
  } catch (e) { /* 下の救済処理へ */ }

  const m = rawText.match(/\{[^{}]*"faq_id"[^{}]*\}/);
  if (m) {
    try {
      const id = Number(JSON.parse(m[0]).faq_id);
      if (Number.isInteger(id) && id >= 0 && id <= 6) return id;
    } catch (e) { /* 何もしない */ }
  }
  return 0;
}
